= Install Rudder agent on RHEL or derivatives

include::{partialsdir}/quick_agent.adoc[]

include::{partialsdir}/rpm_key.adoc[]

include::{partialsdir}/rhel_repo.adoc[]

* on Amazon Linux 1:

----

echo '[Rudder_8.2]
name=Rudder 8.2
baseurl=http://repository.rudder.io/rpm/8.2/RHEL_6/
gpgcheck=1
gpgkey=https://repository.rudder.io/rudder_release_key.pub' > /etc/yum.repos.d/rudder.repo

----

* on Amazon Linux 2:

----

echo '[Rudder_8.2]
name=Rudder 8.2
baseurl=http://repository.rudder.io/rpm/8.2/RHEL_7/
gpgcheck=1
gpgkey=https://repository.rudder.io/rudder_release_key.pub' > /etc/yum.repos.d/rudder.repo

----

* on Amazon Linux 2023:

----

echo '[Rudder_8.2]
name=Rudder 8.2
baseurl=http://repository.rudder.io/rpm/8.2/AL_2023/
gpgcheck=1
gpgkey=https://repository.rudder.io/rudder_release_key.pub' > /etc/yum.repos.d/rudder.repo

----

Install the package:

----

yum install rudder-agent

----
